% ---- Paper 1 ----
\subsection{Modeling Continuous Integration Practice Differences in Industry Software Development \cite{Stahl2014}}
%
\begin{itemize}
    \item \textbf{Found:} using Scopus with query ``continuous integration'' AND ``practice'' AND ``modeling'' AND ``industry''.
    \item \textbf{Justification:} this paper proposes a descriptive model to characterize variation points in how CI practices are implemented across industry projects, including scope, trigger mechanisms, and integration frequency. It provides a foundational vocabulary for reasoning about CI pipeline architecture and design diversity, which is directly relevant to understanding how complex CI pipelines are designed.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Timoth\'{e} Piscaglia.
\end{itemize}

% ---- Paper 2 ----
\subsection{Use and Misuse of Continuous Integration Features: An Empirical Study of Projects That (Mis)Use Travis CI \cite{Gallaba2020}}
%
\begin{itemize}
    \item \textbf{Found:} using Scopus with query ``continuous integration'' AND ``configuration'' AND ``anti-pattern'' AND ``Travis CI''.
    \item \textbf{Justification:} This landmark research examines the usage and misuse of CI configuration feature usage in 9,312 open-source Travis CI projects. It finds four anti-patterns in CI specification files, including 48\% of configuration code used for job environment setup. It also proposes the tool Gretel, which detects anti-patterns in CI configurations, thus addressing the issue of configuration management.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Timoth\'{e} Piscaglia.
\end{itemize}

% ---- Paper 3 ----
\subsection{Configuration Smells in Continuous Delivery Pipelines: A Linter and a Six-Month Study on GitLab \cite{Vassallo2020}}
%
\begin{itemize}
    \item \textbf{Found:} using Scopus with query ``continuous delivery'' AND ``configuration smells'' AND ``pipeline'' AND ``linter''.
    \item \textbf{Justification:} CD-Linter, a semantic linter, is presented in this paper, which identifies four types of configuration smells in pipeline YAML files. The evaluation is done using 5,312 open-source GitLab projects, and the results show 87\% precision and 94\% recall in six months. This research is significant in addressing the problem of pipeline configuration maintainability in complex CI/CD pipelines.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Timoth\'{e} Piscaglia.
\end{itemize}

% ---- Paper 4 ----
\subsection{CI/CD Pipelines Evolution and Restructuring: A Qualitative and Quantitative Study \cite{Zampetti2021}}
%
\begin{itemize}
    \item \textbf{Found:} using Scopus with query ``CI/CD pipeline'' AND ``evolution'' AND ``restructuring''.
    \item \textbf{Justification:} This paper presents a combination of qualitative analysis of 615 commits related to pipeline configuration change and quantitative analysis of 253,492 commits from 4,644 Travis CI projects. The paper provides a taxonomy of 34 CI/CD pipeline restructuring operations and 16 evolution metrics, which directly characterize changes in pipeline component evolution and technical debt accumulation.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Timoth\'{e} Piscaglia.
\end{itemize}

% ---- Paper 5 ----
\subsection{An Empirical Characterization of Bad Practices in Continuous Integration \cite{Zampetti2020}}
%
\begin{itemize}
    \item \textbf{Found:} using Scopus with query ``continuous integration'' AND ``bad practices'' AND ``empirical''.
    \item \textbf{Justification:} In this study, bad practices in CI are empirically examined through interviews with developers, analysis of Stack Overflow, and online surveys. It presents a comprehensive collection of 79 bad smells in CI, grouped into 7 categories, which is the most exhaustive taxonomy of bad practices in CI configurations that are relevant to maintenance. This is important in understanding what makes CI pipelines complex and error-prone.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Timoth\'{e} Piscaglia.
\end{itemize}

% ---- Paper 6 ----
\subsection{How Do Software Developers Use GitHub Actions to Automate Their Workflows? \cite{Kinsman2021}}
%
\begin{itemize}
    \item \textbf{Found:} using Google Scholar with query ``GitHub Actions'' AND ``workflow'' AND ``continuous integration'' AND ``empirical study''.
    \item \textbf{Justification:} This is the first research on the way developers use and create GitHub Actions YAML-based workflow files. It classifies actions based on their purpose, such as CI, deployment, and code quality, and explores the way pipeline-as-code configurations are structured, providing an "inside look" at modern-day CI workflow design.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Pierre-Louis Lef\`{e}vre.
\end{itemize}

% ---- Paper 7 ----
\subsection{An Empirical Study of Architecting for Continuous Delivery and Deployment \cite{Shahin2019}}
%
\begin{itemize}
    \item \textbf{Found:} using Scopus with query ``architecting'' AND ``continuous delivery'' AND ``deployment'' AND ``empirical''.
    \item \textbf{Justification:} in this paper, we propose a conceptual framework for (re-)architecting software systems in support of CD pipelines, based on 21 practitioner interviews and a survey of 91 respondents. It directly addresses the role of software architecture in CD pipelines, including deployment unit granularity and service decomposition, which are critical considerations in designing complex CI pipelines.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Pierre-Louis Lef\`{e}vre.
\end{itemize}

% ---- Paper 8 ----
\subsection{Code Review of Build System Specifications: Prevalence, Purposes, Patterns, and Perceptions \cite{Nejati2023}}
%
\begin{itemize}
    \item \textbf{Found:} using Google Scholar with query ``code review'' AND ``build system'' AND ``specifications'' AND ``continuous integration''.
    \item \textbf{Justification:} This is the first empirical study on code review practices of build system specifications (pipeline-as-code). It analyzed 502,931 change sets from Qt and Eclipse and found that build specs are reviewed twice less frequently than production code, but review comments are more likely to identify defects. This is directly relevant to the assessment of design quality of complex pipeline specifications.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Pierre-Louis Lef\`{e}vre.
\end{itemize}

% ---- Paper 9 ----
\subsection{Automated Reporting of Anti-Patterns and Decay in Continuous Integration \cite{Vassallo2019}}
%
\begin{itemize}
    \item \textbf{Found:} using Scopus with query ``continuous integration'' AND ``anti-patterns'' AND ``decay'' AND ``automated''.
    \item \textbf{Justification:} This paper introduces a new tool, called CI-Odor, for automatically identifying four major anti-patterns of CI, namely Slow Build, Broken Release, Late Merging, and Missing Build Failure Repair, based on build log information and repository information. It illustrates the feasibility of using an automated approach for identifying the degradation of CI pipelines, targeting one of the complexity dimensions of CI.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Pierre-Louis Lef\`{e}vre.
\end{itemize}

% ---- Paper 10 ----
\subsection{Lessons from Eight Years of Operational Data from a Continuous Integration Service: An Exploratory Case Study of CircleCI \cite{Gallaba2022}}
%
\begin{itemize}
    \item \textbf{Found:} using Google Scholar with query ``continuous integration'' AND ``operational data'' AND ``CircleCI'' AND ``case study''.
    \item \textbf{Justification:} In this large-scale empirical study, we utilize eight years of operational data from CircleCI, one of the largest CI service providers. We study the challenges, configurations, and trends in CI service scalability based on thousands of projects, offering a service provider’s point of view on the operation and scaling of CI pipelines in production.
    \item \textbf{Cross-validation:} found by Brice Delcroix, confirmed by Pierre-Louis Lef\`{e}vre.
\end{itemize}

\subsection{Usage, Costs, and Benefits of Continuous Integration in Open-Source Projects \cite{Hilton2016}}
%
\begin{itemize}
    \item \textbf{Found:} using Google Scholar with query ``continuous integration'' AND ``usage'' AND ``benefits'' AND ``open-source'' AND ``empirical''.
    \item \textbf{Justification:} this large-scale empirical study surveys 34,544 open-source projects on GitHub to measure the actual usage of CI systems (primarily Travis CI), the cost in terms of build minutes, and the benefits in terms of reduced merge conflicts and faster pull-request resolution. It provides foundational quantitative evidence that CI adoption is widespread and beneficial, grounding the discussion of why complex pipelines are worth investing in.
    \item \textbf{Cross-validation:} found by Pierre-Louis Lef\`{e}vre, confirmed by Timoth\'{e} Piscaglia.
\end{itemize}

% ---- Paper 27 ----
\subsection{The Impact of Continuous Integration on Other Software Development Practices: A Large-Scale Empirical Study \cite{Zhao2017}}
%
\begin{itemize}
    \item \textbf{Found:} using Scopus with query ``continuous integration'' AND ``impact'' AND ``software development'' AND ``practices'' AND ``empirical''.
    \item \textbf{Justification:} using regression discontinuity design on hundreds of GitHub projects that adopted Travis CI, this study measures how CI uptake causally shifts code writing, pull-request closing, and testing practices. It demonstrates that CI is not an isolated tool choice but reshapes the surrounding development workflow, which is directly relevant to understanding the systemic effects of pipeline design decisions.
    \item \textbf{Cross-validation:} found by Pierre-Louis Lef\`{e}vre, confirmed by Timoth\'{e} Piscaglia.
\end{itemize}

% ---- Paper 28 ----
\subsection{An Empirical Analysis of Build Failures in the Continuous Integration Workflows of Java-Based Open-Source Software \cite{Rausch2017}}
%
\begin{itemize}
    \item \textbf{Found:} using Scopus with query ``continuous integration'' AND ``build failures'' AND ``empirical'' AND ``Java'' AND ``open-source''.
    \item \textbf{Justification:} this paper analyses 4,021 Travis CI builds across 30 Java projects, categorising root causes of build failures into compilation errors, test failures, and environment issues. Understanding the distribution of failure causes is essential for designing resilient pipeline stages and for justifying automated failure-detection mechanisms in complex pipelines.
    \item \textbf{Cross-validation:} found by Pierre-Louis Lef\`{e}vre, confirmed by Timoth\'{e} Piscaglia.
\end{itemize}

% ---- Paper 29 ----
\subsection{Oops, My Tests Broke the Build: An Explorative Analysis of Travis CI with GitHub \cite{Beller2017}}
%
\begin{itemize}
    \item \textbf{Found:} using Google Scholar with query ``Travis CI'' AND ``build breakage'' AND ``test failures'' AND ``GitHub'' AND ``explorative analysis''.
    \item \textbf{Justification:} by linking Travis CI build logs with GitHub event streams, this study shows that most build breaks are caused by test failures rather than compilation errors, and that developers take significantly longer to fix failing builds overnight. These findings directly inform pipeline scheduling and notification design in complex CI environments.
    \item \textbf{Cross-validation:} found by Pierre-Louis Lef\`{e}vre, confirmed by Timoth\'{e} Piscaglia.

% ---- Paper 34 ----
\subsection{The Silent Helper: The Impact of Continuous Integration on Code Reviews \cite{Cassee2020}}
%
\begin{itemize}
    \item \textbf{Found:} using Google Scholar with query ``continuous integration'' AND ``impact'' AND ``code review'' AND ``pull request'' AND ``empirical''.
    \item \textbf{Justification:} by analysing 685 GitHub projects that adopted Travis CI, this paper shows that CI significantly reduces review discussion volume --- saving on average one review comment per pull request --- while the number of code changes per review stays roughly constant. These findings are relevant to understanding how pipeline feedback loops interact with human review processes in complex development workflows.
    \item \textbf{Cross-validation:} found by Pierre-Louis Lef\`{e}vre, confirmed as relevant by Brice Delcroix.
